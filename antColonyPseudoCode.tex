\begin{algorithm}[H]
\caption{Algoritmo Colonia de Hormigas para el MISP}
\KwIn{$n$: número de nodos, $nl$: lista de vecindades, $t$: tiempo máximo (s), $e$: tasa de evaporación, $d$: cantidad de depósito}
\KwOut{$bestSize$: tamaño del mejor conjunto independiente encontrado}

$bestSize \gets 0$\;
$start \gets tiempoActual()$\;
$F[0..n-1] \gets 1$\;
\While{$tiempoActual() - start < t$}{

  \BlankLine
  \tcc{Construcción de la solución}
  \While{exista al menos un nodo $u$ con $F[u] > 0$}{
    $v \gets$ selección aleatoria ponderada por $F$\;
    agregar $v$ a $sol$\;
    $F[v] \gets 0$\;
    \ForEach{$u$ en nl($v$)}{$F[u] \gets 0$}
  }

  \BlankLine
  \tcc{Actualización de la mejor solución}
  \If{$|sol| > bestSize$}{
    $bestSize \gets |sol|$\;

    \BlankLine
    \tcc{Depósito de feromonas}
    \ForEach{$u$ en $sol$}{
      $F[u] \gets F[u] + d$
    }
  }

  \BlankLine
  \tcc{Evaporación de feromonas}
  \For{$i \gets 0$ \KwTo $n-1$}{
    $F[i] \gets \lfloor F[i]\cdot(1-e) \rfloor$
    $F[i] \gets \max(1, F[i])$
  }
}
\Return{$bestSize$}
\end{algorithm}
